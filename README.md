function calculateTotal(items) {
  let sum = 0;
  for (let i = 0; i <= items.length; i++) {

  }
  return sum;

<details>
<summary>🐛 Find the bug</summary>

<p>The loop condition should be <code>i < items.length</code> instead of <code>i <= items.length</code>!</p>

<details>
<summary>✅ See the fixed code</summary>

```javascript
function calculateTotal(items) {
  let sum = 0;
  for (let i = 0; i < items.length; i++) {
    sum += items[i].price;
  }
  return sum;
}
```
